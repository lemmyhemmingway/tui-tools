> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a TUI DevToys clone with 6 developer utilities (JSON, Base64, Hash, UUID, Regex, Timestamp) accessible from a sidebar.

**Architecture:** `App` holds a `Vec<Box<dyn Tool>>` and a `Focus` state. `main.rs` owns the event loop, sidebar rendering, and layout. Each tool in `src/tools/<name>.rs` implements the `Tool` trait with `render`, `handle_key`, and `footer_hints`. Transform logic lives in pure functions for testability.

**Tech Stack:** Rust, ratatui 0.29, crossterm 0.28, serde_json, base64 0.22, sha2/sha1/md-5, uuid, regex, chrono 0.4

```rust
mod textarea;
mod tools;

use crossterm::{
    event::{self, Event, KeyCode, KeyModifiers},
    execute,
    terminal::{disable_raw_mode, enable_raw_mode, EnterAlternateScreen, LeaveAlternateScreen},
};
use ratatui::{
    backend::CrosstermBackend,
    layout::{Constraint, Direction, Layout},
    style::{Color, Modifier, Style},
    text::Span,
    widgets::{Block, Borders, List, ListItem, Paragraph},
    Terminal,
};
use std::io;
use tools::{Action, Focus, Tool};

struct App {
    selected: usize,
    focus: Focus,
    tools: Vec<Box<dyn Tool>>,
}

impl App {
    fn new() -> Self {
        Self {
            selected: 0,
            focus: Focus::Sidebar,
            tools: vec![],
        }
    }
}

fn draw(frame: &mut ratatui::Frame, app: &mut App) {
    let root = Layout::default()
        .direction(Direction::Vertical)
        .constraints([Constraint::Min(0), Constraint::Length(1)])
        .split(frame.area());

    let columns = Layout::default()
        .direction(Direction::Horizontal)
        .constraints([Constraint::Length(18), Constraint::Min(0)])
        .split(root[0]);

    let sidebar_area = columns[0];
    let content_area = columns[1];
    let footer_area = root[1];

    let items: Vec<ListItem> = app.tools.iter().enumerate().map(|(i, t)| {
        let style = if i == app.selected {
            Style::default().fg(Color::Yellow).add_modifier(Modifier::BOLD)
        } else {
            Style::default().fg(Color::White)
        };
        ListItem::new(Span::styled(t.name(), style))
    }).collect();

    frame.render_widget(
        List::new(items).block(Block::default().borders(Borders::ALL).title("tools")),
        sidebar_area,
    );

    if !app.tools.is_empty() {
        let focus = app.focus;
        app.tools[app.selected].render(frame, content_area, focus);
    } else {
        frame.render_widget(
            Paragraph::new("no tools").block(Block::default().borders(Borders::ALL)),
            content_area,
        );
    }

    let hints = if app.focus == Focus::Sidebar {
        "j/k: navigate  Enter: focus tool  q: quit".to_string()
    } else if !app.tools.is_empty() {
        let tool_hints = app.tools[app.selected].footer_hints();
        if tool_hints.is_empty() {
            "Esc: sidebar  ctrl+c: quit".to_string()
        } else {
            format!("Esc: sidebar  {}  ctrl+c: quit", tool_hints)
        }
    } else {
        "q: quit".to_string()
    };
    frame.render_widget(
        Paragraph::new(hints).style(Style::default().fg(Color::DarkGray)),
        footer_area,
    );
}

fn main() -> io::Result<()> {
    enable_raw_mode()?;
    let mut stdout = io::stdout();
    execute!(stdout, EnterAlternateScreen)?;
    let backend = CrosstermBackend::new(stdout);
    let mut terminal = Terminal::new(backend)?;
    let mut app = App::new();

    loop {
        terminal.draw(|f| draw(f, &mut app))?;

        let Event::Key(key) = event::read()? else { continue };

        match app.focus {
            Focus::Sidebar => match key.code {
                KeyCode::Char('q') => break,
                KeyCode::Char('c') if key.modifiers.contains(KeyModifiers::CONTROL) => break,
                KeyCode::Char('j') | KeyCode::Down => {
                    if app.selected + 1 < app.tools.len() {
                        app.selected += 1;
                    }
                }
                KeyCode::Char('k') | KeyCode::Up => {
                    if app.selected > 0 {
                        app.selected -= 1;
                    }
                }
                KeyCode::Enter | KeyCode::Tab => {
                    if !app.tools.is_empty() {
                        app.focus = app.tools[app.selected].initial_focus();
                    }
                }
                _ => {
                    if !app.tools.is_empty() {
                        let focus = app.focus;
                        app.tools[app.selected].handle_key(key, focus);
                    }
                }
            },
            Focus::Input | Focus::Pattern => {
                let focus = app.focus;
                let action = app.tools[app.selected].handle_key(key, focus);
                match action {
                    Action::Quit => break,
                    Action::FocusSidebar => app.focus = Focus::Sidebar,
                    Action::Nothing => {}
                }
            }
        }
    }

    disable_raw_mode()?;
    execute!(terminal.backend_mut(), LeaveAlternateScreen)?;
    Ok(())
}
```

use crate::textarea::TextArea;
use crossterm::event::{KeyCode, KeyEvent, KeyModifiers};
use ratatui::{
    layout::{Constraint, Direction, Layout, Rect},
    style::{Color, Style},
    widgets::{Block, Borders, Paragraph},
    Frame,
};
use serde_json::Value;
use super::{Action, Focus, Tool};

pub struct JsonTool {
    input: TextArea,
    output: String,
    is_error: bool,
}

impl JsonTool {
    pub fn new() -> Self {
        Self { input: TextArea::new(), output: String::new(), is_error: false }
    }
}

fn format_json(input: &str) -> Result<String, String> {
    if input.trim().is_empty() {
        return Ok(String::new());
    }
    let v: Value = serde_json::from_str(input).map_err(|e| e.to_string())?;
    serde_json::to_string_pretty(&v).map_err(|e| e.to_string())
}

impl Tool for JsonTool {
    fn name(&self) -> &'static str { "JSON" }

    fn render(&mut self, frame: &mut Frame, area: Rect, focus: Focus) {
        let chunks = Layout::default()
            .direction(Direction::Vertical)
            .constraints([Constraint::Percentage(50), Constraint::Percentage(50)])
            .split(area);

        self.input.render(frame, chunks[0], matches!(focus, Focus::Input), "Input");

        let output_style = if self.is_error {
            Style::default().fg(Color::Red)
        } else {
            Style::default().fg(Color::Green)
        };
        frame.render_widget(
            Paragraph::new(self.output.clone())
                .style(output_style)
                .block(Block::default().borders(Borders::ALL).title("Output")),
            chunks[1],
        );
    }

    fn handle_key(&mut self, key: KeyEvent, _focus: Focus) -> Action {
        match key.code {
            KeyCode::Esc => return Action::FocusSidebar,
            KeyCode::Char('c') if key.modifiers.contains(KeyModifiers::CONTROL) => {
                return Action::Quit
            }
            _ => {}
        }
        self.input.handle_key(key);
        match format_json(&self.input.content()) {
            Ok(s) => { self.output = s; self.is_error = false; }
            Err(e) => { self.output = e; self.is_error = true; }
        }
        Action::Nothing
    }

    fn footer_hints(&self) -> String { String::new() }
}

        assert!(format_json("not json").is_err());

        assert_eq!(format_json("").unwrap(), "");

}

use crate::textarea::TextArea;
use base64::{engine::general_purpose, Engine as _};
use crossterm::event::{KeyCode, KeyEvent, KeyModifiers};
use ratatui::{
    layout::{Constraint, Direction, Layout, Rect},
    style::{Color, Style},
    widgets::{Block, Borders, Paragraph},
    Frame,
};
use super::{Action, Focus, Tool};

#[derive(Clone, Copy, PartialEq)]
enum Mode { Encode, Decode }

pub struct Base64Tool {
    input: TextArea,
    output: String,
    is_error: bool,
    mode: Mode,
}

impl Base64Tool {
    pub fn new() -> Self {
        Self { input: TextArea::new(), output: String::new(), is_error: false, mode: Mode::Encode }
    }

    fn transform(&mut self) {
        let content = self.input.content();
        match self.mode {
            Mode::Encode => {
                self.output = encode_base64(&content);
                self.is_error = false;
            }
            Mode::Decode => match decode_base64(&content) {
                Ok(s) => { self.output = s; self.is_error = false; }
                Err(e) => { self.output = e; self.is_error = true; }
            }
        }
    }
}

fn encode_base64(input: &str) -> String {
    general_purpose::STANDARD.encode(input.as_bytes())
}

fn decode_base64(input: &str) -> Result<String, String> {
    let bytes = general_purpose::STANDARD.decode(input.trim()).map_err(|e| e.to_string())?;
    String::from_utf8(bytes).map_err(|e| e.to_string())
}

impl Tool for Base64Tool {
    fn name(&self) -> &'static str { "Base64" }

    fn render(&mut self, frame: &mut Frame, area: Rect, focus: Focus) {
        let chunks = Layout::default()
            .direction(Direction::Vertical)
            .constraints([Constraint::Percentage(50), Constraint::Percentage(50)])
            .split(area);

        let mode_label = match self.mode { Mode::Encode => "Input (encode)", Mode::Decode => "Input (decode)" };
        self.input.render(frame, chunks[0], matches!(focus, Focus::Input), mode_label);

        let output_style = if self.is_error { Style::default().fg(Color::Red) } else { Style::default().fg(Color::Green) };
        frame.render_widget(
            Paragraph::new(self.output.clone())
                .style(output_style)
                .block(Block::default().borders(Borders::ALL).title("Output")),
            chunks[1],
        );
    }

    fn handle_key(&mut self, key: KeyEvent, _focus: Focus) -> Action {
        match key.code {
            KeyCode::Esc => return Action::FocusSidebar,
            KeyCode::Char('c') if key.modifiers.contains(KeyModifiers::CONTROL) => return Action::Quit,
            KeyCode::Tab => {
                self.mode = if self.mode == Mode::Encode { Mode::Decode } else { Mode::Encode };
                self.transform();
                return Action::Nothing;
            }
            _ => {}
        }
        self.input.handle_key(key);
        self.transform();
        Action::Nothing
    }

    fn footer_hints(&self) -> String {
        "Tab: toggle encode/decode".to_string()
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_encode() {
        assert_eq!(encode_base64("hello"), "aGVsbG8=");
    }

    #[test]
    fn test_decode_valid() {
        assert_eq!(decode_base64("aGVsbG8=").unwrap(), "hello");
    }

    #[test]
    fn test_decode_invalid() {
        assert!(decode_base64("!!!").is_err());
    }
}

use crate::textarea::TextArea;
use crossterm::event::{KeyCode, KeyEvent, KeyModifiers};
use md5::{Md5, Digest as _};
use ratatui::{
    layout::{Constraint, Direction, Layout, Rect},
    style::{Color, Style},
    widgets::{Block, Borders, Paragraph},
    Frame,
};
use sha1::Sha1;
use sha2::Sha256;
use super::{Action, Focus, Tool};

pub struct HashTool {
    input: TextArea,
    output: String,
}

impl HashTool {
    pub fn new() -> Self {
        Self { input: TextArea::new(), output: String::new() }
    }
}

fn compute_hashes(input: &str) -> String {
    if input.is_empty() { return String::new(); }
    let bytes = input.as_bytes();

    let md5_hex = {
        let mut h = Md5::new();
        h.update(bytes);
        format!("{:x}", h.finalize())
    };
    let sha1_hex = {
        use sha1::Digest;
        let mut h = Sha1::new();
        h.update(bytes);
        format!("{:x}", h.finalize())
    };
    let sha256_hex = {
        use sha2::Digest;
        let mut h = Sha256::new();
        h.update(bytes);
        format!("{:x}", h.finalize())
    };

    format!("MD5:    {}\nSHA1:   {}\nSHA256: {}", md5_hex, sha1_hex, sha256_hex)
}

impl Tool for HashTool {
    fn name(&self) -> &'static str { "Hash" }

    fn render(&mut self, frame: &mut Frame, area: Rect, focus: Focus) {
        let chunks = Layout::default()
            .direction(Direction::Vertical)
            .constraints([Constraint::Percentage(50), Constraint::Percentage(50)])
            .split(area);

        self.input.render(frame, chunks[0], matches!(focus, Focus::Input), "Input");

        frame.render_widget(
            Paragraph::new(self.output.clone())
                .style(Style::default().fg(Color::Green))
                .block(Block::default().borders(Borders::ALL).title("Hashes")),
            chunks[1],
        );
    }

    fn handle_key(&mut self, key: KeyEvent, _focus: Focus) -> Action {
        match key.code {
            KeyCode::Esc => return Action::FocusSidebar,
            KeyCode::Char('c') if key.modifiers.contains(KeyModifiers::CONTROL) => return Action::Quit,
            _ => {}
        }
        self.input.handle_key(key);
        self.output = compute_hashes(&self.input.content());
        Action::Nothing
    }

    fn footer_hints(&self) -> String { String::new() }
}

mod tests {
    use super::*;

    #[test]
    fn test_hash_hello() {
        let out = compute_hashes("hello");
        assert!(out.contains("MD5:    5d41402abc4b2a76b9719d911017c592"));
        assert!(out.contains("SHA1:   aaf4c61ddcc5e8a2dabede0f3b482cd9aea9434d"));
        assert!(out.contains("SHA256: 2cf24dba5fb0a30e26e83b2ac5b9e29e1b161e5c1fa7425e73043362938b9824"));
    }

    #[test]

        assert!(out.is_empty());
    }
}

use crossterm::event::{KeyCode, KeyEvent, KeyModifiers};
use ratatui::{
    layout::Rect,
    style::{Color, Modifier, Style},
    text::{Line, Span},
    widgets::{Block, Borders, Paragraph},
    Frame,

use uuid::Uuid;

use crate::textarea::TextArea;
use crossterm::event::{KeyCode, KeyEvent, KeyModifiers};
use ratatui::{
    layout::{Constraint, Direction, Layout, Rect},
    style::{Color, Style},
    widgets::{Block, Borders, Paragraph},
    Frame,
};
use regex::Regex;
use super::{Action, Focus, Tool};

pub struct RegexTool {
    pattern: TextArea,
    input: TextArea,
    output: String,
    is_error: bool,
}

impl RegexTool {
    pub fn new() -> Self {
        Self {
            pattern: TextArea::new(),
            input: TextArea::new(),
            output: String::new(),
            is_error: false,
        }
    }

    fn transform(&mut self) {
        let pat = self.pattern.content();
        let text = self.input.content();
        match apply_regex(&pat, &text) {
            Ok(s) => { self.output = s; self.is_error = false; }
            Err(e) => { self.output = e; self.is_error = true; }
        }
    }
}

fn apply_regex(pattern: &str, text: &str) -> Result<String, String> {
    if pattern.is_empty() { return Ok(text.to_string()); }
    let re = Regex::new(pattern).map_err(|e| e.to_string())?;
    let result = re.replace_all(text, |caps: &regex::Captures| {
        format!("[{}]", &caps[0])
    });
    Ok(result.to_string())
}

impl Tool for RegexTool {
    fn name(&self) -> &'static str { "Regex" }

    fn initial_focus(&self) -> Focus { Focus::Pattern }

    fn render(&mut self, frame: &mut Frame, area: Rect, focus: Focus) {
        let chunks = Layout::default()
            .direction(Direction::Vertical)
            .constraints([Constraint::Length(3), Constraint::Percentage(50), Constraint::Min(0)])
            .split(area);

        self.pattern.render(frame, chunks[0], matches!(focus, Focus::Pattern), "Pattern");
        self.input.render(frame, chunks[1], matches!(focus, Focus::Input), "Test Text");

        let output_style = if self.is_error { Style::default().fg(Color::Red) } else { Style::default().fg(Color::Green) };
        frame.render_widget(
            Paragraph::new(self.output.clone())
                .style(output_style)
                .block(Block::default().borders(Borders::ALL).title("Matches")),
            chunks[2],
        );
    }

    fn handle_key(&mut self, key: KeyEvent, focus: Focus) -> Action {
        match key.code {
            KeyCode::Esc => return Action::FocusSidebar,
            KeyCode::Char('c') if key.modifiers.contains(KeyModifiers::CONTROL) => return Action::Quit,
            KeyCode::Tab => {
                return match focus {
                    Focus::Pattern => Action::FocusSidebar, // caller sets focus=Input after this won't work directly
                    _ => Action::FocusSidebar,
                };
            }
            _ => {}
        }
        match focus {
            Focus::Pattern => self.pattern.handle_key_single(key),
            Focus::Input => self.input.handle_key(key),
            Focus::Sidebar => {}
        }
        self.transform();
        Action::Nothing
    }

    fn footer_hints(&self) -> String {
        "Tab: switch pattern/text".to_string()
    }
}

use crate::textarea::TextArea;
use chrono::{DateTime, TimeZone, Utc};
use crossterm::event::{KeyCode, KeyEvent, KeyModifiers};
use ratatui::{
    layout::{Constraint, Direction, Layout, Rect},
    style::{Color, Style},
    widgets::{Block, Borders, Paragraph},
    Frame,
};
use super::{Action, Focus, Tool};

pub struct TimestampTool {
    input: TextArea,
    output: String,
    is_error: bool,
}

impl TimestampTool {
    pub fn new() -> Self {
        Self { input: TextArea::new(), output: String::new(), is_error: false }
    }
}

fn convert_timestamp(input: &str) -> Result<String, String> {
    let trimmed = input.trim();
    if trimmed.is_empty() { return Ok(String::new()); }

    // Try parsing as unix integer first
    if let Ok(ts) = trimmed.parse::<i64>() {
        let dt = Utc.timestamp_opt(ts, 0)
            .single()
            .ok_or_else(|| "timestamp out of range".to_string())?;
        return Ok(format!(
            "UTC:   {}\nLocal: {}",
            dt.format("%Y-%m-%d %H:%M:%S UTC"),
            dt.format("%Y-%m-%d %H:%M:%S UTC"),
        ));
    }

    // Try RFC3339 / ISO8601
    if let Ok(dt) = DateTime::parse_from_rfc3339(trimmed) {
        return Ok(format!(
            "Unix: {}\nUTC:  {}",
            dt.timestamp(),
            dt.with_timezone(&Utc).format("%Y-%m-%d %H:%M:%S UTC"),
        ));
    }

    Err("unrecognised input — try a unix timestamp (e.g. 1700000000) or RFC3339 date (e.g. 2023-11-14T22:13:20Z)".to_string())
}

impl Tool for TimestampTool {
    fn name(&self) -> &'static str { "Timestamp" }

    fn render(&mut self, frame: &mut Frame, area: Rect, focus: Focus) {
        let chunks = Layout::default()
            .direction(Direction::Vertical)
            .constraints([Constraint::Percentage(50), Constraint::Percentage(50)])
            .split(area);

        self.input.render(frame, chunks[0], matches!(focus, Focus::Input), "Input (unix or RFC3339)");

        let output_style = if self.is_error { Style::default().fg(Color::Red) } else { Style::default().fg(Color::Green) };
        frame.render_widget(
            Paragraph::new(self.output.clone())
                .style(output_style)
                .block(Block::default().borders(Borders::ALL).title("Output")),
            chunks[1],
        );
    }

    fn handle_key(&mut self, key: KeyEvent, _focus: Focus) -> Action {
        match key.code {
            KeyCode::Esc => return Action::FocusSidebar,
            KeyCode::Char('c') if key.modifiers.contains(KeyModifiers::CONTROL) => return Action::Quit,
            _ => {}
        }
        self.input.handle_key(key);
        match convert_timestamp(&self.input.content()) {
            Ok(s) => { self.output = s; self.is_error = false; }
            Err(e) => { self.output = e; self.is_error = true; }
        }
        Action::Nothing
    }

    fn footer_hints(&self) -> String { String::new() }
}
